these is the aggregate function for finding the amount in cart 


db.getCollection('carts').aggregate(
  [
    { $unwind: { path: '$items' } },
    {
      $lookup: {
        from: 'products',
        localField: 'items.product',
        foreignField: '_id',
        as: 'items.product'
      }
    },
    { $unwind: { path: '$items.product' } },
    {
      $unwind: { path: '$items.product.variants' }
    },
    {
      $match: {
        $expr: {
          $eq: [
            '$items.variant',
            '$items.product.variants._id'
          ]
        }
      }
    },
    {
      $addFields: {
        itemPrice: {
          $multiply: [
            '$items.quantity',
            '$items.product.variants.price.amount'
          ]
        }
      }
    },
    {
      $group: {
        _id: '$_id',
        total: { $sum: '$itemPrice' },
        items: { $push: '$items' }
      }
    }
  ],
  { maxTimeMS: 60000, allowDiskUse: true }
);
